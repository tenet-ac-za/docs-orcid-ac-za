## Start Here

Use this page as a "course in a box" kit:

1. Work through each module in order.
2. Use the activity prompts to run mini workshops with your team.
3. Adapt the facilitator notes for your local context.

### Learning Outcomes

By the end of this course, participants should be able to:

- Explain ORCID value to researchers, institutions, and libraries.
- Support creation and maintenance of high-quality ORCID records.
- Guide ORCID integration into institutional systems and workflows.
- Handle common adoption questions and troubleshooting scenarios.

## Delivery Guide

- Run as self-paced learning over one week, or deliver as a half-day workshop.
- Begin each module with one local example to make the content immediately relevant.
- Keep activities lightweight and discussion-based.
- Document questions that recur and convert them into FAQ material.

<script>
  (function () {
    var storageKey = "orcid-course-progress-v1";
    var checkboxes = document.querySelectorAll("input[type='checkbox'][data-progress-key]");
    var resetButton = document.querySelector("[data-progress-reset]");

    if (!checkboxes.length || typeof window.localStorage === "undefined") {
      return;
    }

    var state = {};
    try {
      state = JSON.parse(window.localStorage.getItem(storageKey) || "{}");
    } catch (err) {
      state = {};
    }

    checkboxes.forEach(function (box) {
      var key = box.getAttribute("data-progress-key");
      box.checked = Boolean(state[key]);
      box.addEventListener("change", function () {
        state[key] = box.checked;
        window.localStorage.setItem(storageKey, JSON.stringify(state));
      });
    });

    if (resetButton) {
      resetButton.addEventListener("click", function () {
        checkboxes.forEach(function (box) {
          box.checked = false;
        });
        window.localStorage.removeItem(storageKey);
      });
    }

    var quizStorageKey = "orcid-course-quiz-v1";
    var quizQuestions = document.querySelectorAll(".quiz-question");
    var quizCheck = document.querySelector("[data-quiz-check]");
    var quizReset = document.querySelector("[data-quiz-reset]");
    var quizResult = document.querySelector("[data-quiz-result]");

    function getFeedbackNode(question) {
      var node = question.querySelector(".quiz-feedback");
      if (!node) {
        node = document.createElement("p");
        node.className = "quiz-feedback";
        node.setAttribute("aria-live", "polite");
        question.appendChild(node);
      }
      return node;
    }

    function clearQuestionState(question) {
      question.classList.remove("is-correct");
      question.classList.remove("is-incorrect");
      question.classList.remove("is-unanswered");
      var node = getFeedbackNode(question);
      node.textContent = "";
    }

    function saveQuizState() {
      var quizState = {};
      quizQuestions.forEach(function (question, index) {
        var selected = question.querySelector("input[type='radio']:checked");
        if (selected) {
          quizState["q" + index] = selected.value;
        }
      });
      window.localStorage.setItem(quizStorageKey, JSON.stringify(quizState));
    }

    function loadQuizState() {
      var quizState = {};
      try {
        quizState = JSON.parse(window.localStorage.getItem(quizStorageKey) || "{}");
      } catch (err) {
        quizState = {};
      }

      quizQuestions.forEach(function (question, index) {
        var value = quizState["q" + index];
        if (value) {
          var option = question.querySelector("input[type='radio'][value='" + value + "']");
          if (option) {
            option.checked = true;
          }
        }
      });
    }

    if (quizQuestions.length) {
      loadQuizState();
      quizQuestions.forEach(function (question) {
        clearQuestionState(question);
        var radios = question.querySelectorAll("input[type='radio']");
        radios.forEach(function (radio) {
          radio.addEventListener("change", function () {
            clearQuestionState(question);
            saveQuizState();
          });
        });
      });
    }

    if (quizCheck && quizQuestions.length && quizResult) {
      quizCheck.addEventListener("click", function () {
        var correct = 0;
        var answered = 0;

        quizQuestions.forEach(function (question) {
          clearQuestionState(question);
          var selected = question.querySelector("input[type='radio']:checked");
          var answer = question.getAttribute("data-answer");
          var node = getFeedbackNode(question);
          if (selected) {
            answered += 1;
            if (selected.value === answer) {
              correct += 1;
              question.classList.add("is-correct");
              node.textContent = "Correct.";
            } else {
              question.classList.add("is-incorrect");
              node.textContent = "Not quite. Review this lesson and try again.";
            }
          } else {
            question.classList.add("is-unanswered");
            node.textContent = "Please select an answer.";
          }
        });

        if (answered < quizQuestions.length) {
          quizResult.textContent = "Please answer all questions before checking your score.";
          return;
        }

        var scoreText = "You scored " + correct + " out of " + quizQuestions.length + ".";
        if (correct === quizQuestions.length) {
          quizResult.textContent = scoreText + " Excellent work.";
        } else if (correct >= 2) {
          quizResult.textContent = scoreText + " Good progress. Review Module 1 for a full score.";
        } else {
          quizResult.textContent = scoreText + " Revisit the lesson pages and try again.";
        }
      });
    }

    if (quizReset && quizQuestions.length && quizResult) {
      quizReset.addEventListener("click", function () {
        quizQuestions.forEach(function (question) {
          clearQuestionState(question);
          var radios = question.querySelectorAll("input[type='radio']");
          radios.forEach(function (radio) {
            radio.checked = false;
          });
        });
        quizResult.textContent = "";
        window.localStorage.removeItem(quizStorageKey);
      });
    }
  })();
</script>
